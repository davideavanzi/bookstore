Table interview {
  id int [pk]
  title varchar [not null]
  content text [not null]
  interviewer varchar 
  book_id int [ref: > book.id]
}

Table book {
  id int [pk]
  title varchar
  cover varchar
  abstract text
  fact_sheet text
  genre_id int [ref: > genre.id]
}


Table author {
  id int [pk]
  name varchar
  bio text
  photo varchar
}

Table book_author{
  id_book int [ref: > book.id]
  id_author int [ref: > author.id]

}

Table event {
  id int [pk]
  title varchar
  id_book int [ref: > book.id]
}

Table review {
  id int [pk]
  star int
  title varchar
  content text
  id_user int [ref: > user.id]
}

Table user {
  id int [pk]
  name varchar
  email varchar
  phone varchar
  address varchar
  password varchar
}

Table cart {
  id int [pk]
  expiration_date date
  user_id int [ref: > user.id]
}

Table cart_book {
    cart_id int [ref: > cart.id]
    book_id int [ref: > book.id]

}

Table reservation {
  id int [pk]
  amount int
  book_id int [ref: > book.id]
  user_id int [ref: > user.id]

}

Table  genre {
  id int [pk]
  name varchar
  
}

Table theme_book {
  book_id int [ref: > book.id]
  theme_id int [ref: > theme.id]

}

Table theme {
  id int [pk]
  name varchar
}
